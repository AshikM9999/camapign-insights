        // Helper function to categorize responses

        function getResponseCategory(responses) {

            if (responses >= 1 && responses <= 10) {

                return "1-10";

            } else if (responses >= 11 && responses <= 100) {

                return "11-100";

            } else if (responses >= 101 && responses <= 1000) {

                return "101-1000";

            }

            return "Other"; // Fallback for values outside defined ranges

        }

        // Function to handle the search/filter logic

        function filterCampaigns() {

            const selectedRegion = document.getElementById('region-select').value;

            const selectedResponsesCategory = document.getElementById('responses-select').value;


            let tempFilteredCampaigns = allCampaigns;


            // Filter by Region

            if (selectedRegion !== 'All') {

                tempFilteredCampaigns = tempFilteredCampaigns.filter(

                    campaign => campaign.Region === selectedRegion

                );

            }


            // Filter by Responses in Campaign category

            if (selectedResponsesCategory !== 'All') {

                tempFilteredCampaigns = tempFilteredCampaigns.filter(campaign =>

                    getResponseCategory(campaign["Responses in Campaign"]) === selectedResponsesCategory

                );

            }


            filteredCampaigns = tempFilteredCampaigns;

            renderCampaignTable(filteredCampaigns);

            renderCharts(filteredCampaigns);

        }

            filterCampaigns(); // Initial render of all campaigns
